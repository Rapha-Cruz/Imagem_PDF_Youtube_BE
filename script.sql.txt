-- Cria a tabela 'produtos' no schema público
CREATE TABLE produtos (
    -- Chave primária auto-incrementável
    id SERIAL PRIMARY KEY,    
    -- Nome do produto. Não pode ser nulo.
    nome VARCHAR(60) NOT NULL,    
    -- URL pública da imagem do produto.
    imagem_url VARCHAR(512),    
    -- URL pública do documento PDF do produto.
    pdf_url VARCHAR(512),    
    -- Link do vídeo do YouTube (formatado como 'embed').
    youtube_link VARCHAR(512),
    -- Timestamp da criação (opcional, mas recomendado)
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
